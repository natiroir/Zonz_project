void world_paint_spot(int x, int y, int player_id, t_player players[MAX_PLAYERS]) {
    if (x >= 0 && x < MAP_SIZE && y >= 0 && y < MAP_SIZE) {
        int index = y * MAP_SIZE + x;
        int old_player_id = mapmem[index];  // ğŸ” ID du joueur prÃ©cÃ©dent

        // Si le pixel Ã©tait dÃ©jÃ  peint par un autre joueur, on rÃ©duit son compteur
        if (old_player_id >= 0 && old_player_id < MAX_PLAYERS && old_player_id != player_id) {
            players[old_player_id].count--; 
        }

        // Peindre le pixel avec le nouvel ID et mettre Ã  jour le compteur du joueur actuel
        mapmem[index] = player_id;
        players[player_id].count++;  // âœ… Ajoute un pixel au nouveau joueur
    }
}
