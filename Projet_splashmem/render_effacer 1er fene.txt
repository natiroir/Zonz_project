int display_score_window(t_player *players[MAX_PLAYERS]) 
{
    uint16_t window_width = 400;
    uint16_t window_height = 350;

    // CrÃ©er la fenÃªtre et le renderer
    SDL_Window *score_window = SDL_CreateWindow("RÃ©sultats de la Partie", SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED, window_width, window_height, SDL_WINDOW_SHOWN);
    SDL_Renderer *renderer = SDL_CreateRenderer(score_window, -1, SDL_RENDERER_ACCELERATED);
    TTF_Font *font = TTF_OpenFont("arial.ttf", 24);

    if (!score_window || !renderer || !font) {
        printf("Erreur crÃ©ation fenÃªtre scores : %s\n", SDL_GetError());
        return 0;
    }

    // RÃ©cupÃ©rer l'ID de la fenÃªtre des scores
    uint32_t score_window_id = SDL_GetWindowID(score_window);

    // DÃ©finition du bouton "Rejouer"
    SDL_Rect button = {window_width / 2 - 75, window_height - 80, 150, 50};

    SDL_Event event;
    int running = 1;
    int restart_game = 0;  // ðŸ”¹ Flag pour savoir si on doit redÃ©marrer la partie

    while (running) {
        while (SDL_PollEvent(&event)) {
            if (event.type == SDL_WINDOWEVENT && event.window.event == SDL_WINDOWEVENT_CLOSE && event.window.windowID == score_window_id) {
                printf("ðŸ”´ Fermeture demandÃ©e pour la fenÃªtre des scores\n");
                running = 0;
            }
            if (event.type == SDL_KEYDOWN && event.key.keysym.sym == SDLK_ESCAPE) {
                printf("ðŸ”µ Fermeture de la fenÃªtre des scores via Ã‰chap\n");
                running = 0;
            }
            if (event.type == SDL_MOUSEBUTTONDOWN) {
                int x, y;
                SDL_GetMouseState(&x, &y);
                if (x >= button.x && x <= button.x + button.w && y >= button.y && y <= button.y + button.h) {
                    printf("ðŸ”„ Rejouer bouton cliquÃ© !\n");
                    restart_game = 1;  // ðŸ”¹ Indiquer qu'on veut effacer la fenÃªtre principale
                    running = 0;
                }
            }
        }

        SDL_Delay(100);
        SDL_SetRenderDrawColor(renderer, 255, 255, 255, 255);
        SDL_RenderClear(renderer);

        // Dessiner le bouton "Rejouer"
        SDL_SetRenderDrawColor(renderer, 0, 0, 255, 255);
        SDL_RenderFillRect(renderer, &button);

        // Afficher "Rejouer"
        SDL_Color white = {255, 255, 255, 255};
        SDL_Surface *text_surface = TTF_RenderText_Solid(font, "Rejouer", white);
        if (text_surface) {
            SDL_Texture *text_texture = SDL_CreateTextureFromSurface(renderer, text_surface);
            SDL_Rect text_rect = {
                button.x + (button.w / 2) - (text_surface->w / 2),
                button.y + (button.h / 2) - (text_surface->h / 2),
                text_surface->w, text_surface->h
            };
            SDL_RenderCopy(renderer, text_texture, NULL, &text_rect);
            SDL_FreeSurface(text_surface);
            SDL_DestroyTexture(text_texture);
        }

        SDL_RenderPresent(renderer);
    }

    // âœ… Fermer uniquement la fenÃªtre des scores
    TTF_CloseFont(font);
    SDL_DestroyRenderer(renderer);
    SDL_DestroyWindow(score_window);

    // âœ… Retourner 1 si on veut redÃ©marrer le jeu, sinon 0
    return restart_game;
}
